def disemvowel(string):
    vowels = ['a','e','i','o','u']
    str1 = ''
    
    for c in string:
        if c not in vowels:
            str1 += c
            
    return(str1)

```

def oddball_sum(numbers):
    ''' Take in an array of integers and returns the sum of all the odd elements '''
    thesum = 0
    
    for num in numbers:
        if num % 2 == 1:
            thesum += num
            
    return(thesum)
```


```python
oddball_sum([1,2,3,4,5])
```




    9




```python

```

import numpy as np

def lucky_sevens(numbers):
    ''' takes in an array of integers and returns true if any three consecutive elements sum to 7 '''
    
    print(numbers) 
    
    a = np.array(numbers)

    
    for i in range(0, a.shape[0]): 
        thesum = np.sum(a[i:i+3])
        print(thesum)
        
        if thesum == 7:
            return True
        
    return False
